    <script>

        // Pricing Data

        const planDetails = {

            "SV Base": { lensType: "Standard Single Vision", monthlyFee: 30, totalFee: 360, allowedMaterials: ["Plastic 1.49", "Polycarbonate"], allowedOptions: ["Clear"] },

            "SV Plus": { lensType: "Single Vision Digital", monthlyFee: 40, totalFee: 480, allowedMaterials: ["Plastic 1.49", "Polycarbonate", "1.67", "1.74"], allowedOptions: ["Clear", "Blue Light Filter", "Transitions Gen S"] },

            "SV Deluxe": { lensType: "Eyezen Start", monthlyFee: 50, totalFee: 600, allowedMaterials: ["Polycarbonate", "1.67", "1.74"], includedOptions: ["Blue Light Filter", "Transitions Gen S"] },

            "MF Base": { lensType: "Comfort Max", monthlyFee: 55, totalFee: 660, allowedMaterials: ["Plastic 1.49", "Polycarbonate"], allowedOptions: ["Clear"] },

            "MF Plus": { lensType: "Physio", monthlyFee: 70, totalFee: 840, allowedMaterials: ["Plastic 1.49", "Polycarbonate", "1.67", "1.74"], allowedOptions: ["Clear", "Blue Light Filter", "Transitions Gen S"] },

            "MF Deluxe": { lensType: "XR Pro", monthlyFee: 90, totalFee: 1080, allowedMaterials: ["Polycarbonate", "1.67", "1.74"], includedOptions: ["Blue Light Filter", "Transitions Gen S"] }

        };


        const standalonePricing = {

            "Standard Single Vision": { "Plastic 1.49": 230, "Polycarbonate": 310 },

            "Single Vision Digital": { "Plastic 1.49": 250, "Polycarbonate": 330 },

            "Eyezen Start": { "Polycarbonate": 350, "1.67": 430 },

            "Comfort Max": { "Plastic 1.49": 480, "Polycarbonate": 560 },

            "Physio": { "Plastic 1.49": 500, "Polycarbonate": 580 },

            "XR Pro": { "Polycarbonate": 600, "1.67": 680 }

        };


        const extraCosts = {

            "Blue Light Filter": 50,

            "Transitions Gen S": 150,

            "1.74": 100

        };


        // DOM Elements

        const planType = document.getElementById("planType");

        const lensType = document.getElementById("lensType");

        const lensMaterial = document.getElementById("lensMaterial");

        const lensOption = document.getElementById("lensOption");

        const frameCost = document.getElementById("frameCost");

        const raaMember = document.getElementById("raaMember");


        const monthlyPlanCost = document.getElementById("monthlyPlanCost");

        const totalPlanCost = document.getElementById("totalPlanCost");

        const planExtrasCost = document.getElementById("planExtrasCost");

        const lensPriceDisplay = document.getElementById("lensPrice");

        const frameCostDisplay = document.getElementById("frameCostDisplay");

        const totalStandaloneCost = document.getElementById("totalStandaloneCost");

        const savings = document.getElementById("savings");

        const extrasBreakdown = document.getElementById("extrasBreakdown");


        // Update Plan Details

        function updatePlanDetails() {

            const selectedPlan = planType.value;

            const plan = planDetails[selectedPlan];


            // Update Lens Type

            lensType.innerHTML = `<option value="${plan.lensType}">${plan.lensType}</option>`;


            // Update Allowed Materials

            lensMaterial.innerHTML = "";

            plan.allowedMaterials.forEach(material => {

                const option = document.createElement("option");

                option.value = material;

                option.textContent = material;

                lensMaterial.appendChild(option);

            });


            // Update Allowed Options

            lensOption.innerHTML = "";

            if (plan.allowedOptions) {

                plan.allowedOptions.forEach(option => {

                    const opt = document.createElement("option");

                    opt.value = option;

                    opt.textContent = option;

                    lensOption.appendChild(opt);

                });

            } else {

                plan.includedOptions.forEach(option => {

                    const opt = document.createElement("option");

                    opt.value = option;

                    opt.textContent = option;

                    lensOption.appendChild(opt);

                });

            }


            // Update Plan Costs

            monthlyPlanCost.textContent = `$${plan.monthlyFee}`;

            totalPlanCost.textContent = `$${plan.totalFee}`;


            calculateCosts();

        }


        // Calculate Costs

        function calculateCosts() {

            const selectedLensType = lensType.value;

            const selectedMaterial = lensMaterial.value;

            const selectedOption = lensOption.value;

            const framePrice = parseFloat(frameCost.value) || 0;

            const isRAA = raaMember.checked;


            // Base Lens Price (Standalone Pricing)

            let lensBasePrice = standalonePricing[selectedLensType][selectedMaterial];

            if (isRAA) {

                lensBasePrice *= 0.8; // Apply RAA Discount

            }


            // Add Extra Costs (if any)

            let extraCost = 0;

            const plan = planDetails[planType.value];

            if (!plan.includedOptions?.includes(selectedOption) && selectedOption in extraCosts) {

                extraCost = extraCosts[selectedOption];

            }

            if (selectedMaterial === "1.74") {

                extraCost += extraCosts["1.74"];

            }


            const totalStandalone = lensBasePrice + framePrice + extraCost;


            // Update Standalone Costs

            lensPriceDisplay.textContent = `$${lensBasePrice.toFixed(2)}`;

            frameCostDisplay.textContent = `$${framePrice.toFixed(2)}`;

            totalStandaloneCost.textContent = `$${totalStandalone.toFixed(2)}`;


            // Update Vision Plan Extras Cost

            planExtrasCost.textContent = `$${extraCost.toFixed(2)}`;


            // Calculate Savings

            const planCost = plan.totalFee + extraCost;

            const savingsAmount = totalStandalone - planCost;


            savings.innerHTML = savingsAmount > 0

                ? `<p>You save $${Math.abs(savingsAmount).toFixed(2)} with the Vision Plan.</p>`

                : `<p>You save $${Math.abs(savingsAmount).toFixed(2)} with the Standalone Purchase.</p>`;


            // Extras Breakdown

            extrasBreakdown.innerHTML = extraCost > 0

                ? `<p>Added Extra: ${selectedOption} - $${extraCost.toFixed(2)}</p>`

                : `<p>No additional extras selected.</p>`;

        }


        // Event Listeners

        planType.addEventListener("change", updatePlanDetails);

        lensMaterial.addEventListener("change", calculateCosts);

        lensOption.addEventListener("change", calculateCosts);

        frameCost.addEventListener("input", calculateCosts);

        raaMember.addEventListener("change", calculateCosts);


        // Initialize Tool

        updatePlanDetails();

    </script>
